53 int SensorCapture::enumerateDevices()
54 {
55  mSlDevPid.clear();
56  mSlDevFwVer.clear();
57 
58  struct hid_device_info *devs, *cur_dev;
59 
60  if (hid_init()==-1)
61  return 0;
62 
63  devs = hid_enumerate(SL_USB_VENDOR, 0x0);
64  cur_dev = devs;
65  while (cur_dev) {
66  int fw_major = cur_dev->release_number>>8;
67  int fw_minor = cur_dev->release_number&0x00FF;
68  uint16_t pid = cur_dev->product_id;
69  std::string sn_str = wstr2str( cur_dev->serial_number );
70  int sn = std::stoi( sn_str );
71 
72  mSlDevPid[sn]=pid;
73  mSlDevFwVer[sn]=cur_dev->release_number;
74 
75  if(mVerbose)
76  {
77  std::ostringstream smsg;
78 
79  smsg << "Device Found: " << std::endl;
80  smsg << " VID: " << std::hex << cur_dev->vendor_id << " PID: " << std::hex << cur_dev->product_id << std::endl;
81  smsg << " Path: " << cur_dev->path << std::endl;
82  smsg << " Serial_number: " << sn_str << std::endl;
83  smsg << " Manufacturer: " << wstr2str(cur_dev->manufacturer_string) << std::endl;
84  smsg << " Product: " << wstr2str(cur_dev->product_string) << std::endl;
85  smsg << " Release number: v" << fw_major << "." << fw_minor << std::endl;
86  smsg << "***" << std::endl;
87 
88  INFO_OUT(mVerbose,smsg.str());
89  }
90 
91  cur_dev = cur_dev->next;
92  }
93 
94  hid_free_enumeration(devs);
95 
96  return mSlDevPid.size();
97 }

113 bool SensorCapture::initializeSensors( int sn )
114 {
115  if(mSlDevPid.size()==0)
116  {
117  enumerateDevices();
118  }
119 
120  std::string sn_str;
121 
122  if(sn!=-1)
123  sn_str = std::to_string(sn);
124  else
125  {
126  if(mSlDevPid.size()==0)
127  {
128  enumerateDevices();
129  }
130 
131  if(mSlDevPid.size()==0)
132  {
133  ERROR_OUT(mVerbose,"No available ZED Mini or ZED2 cameras");
134  return false;
135  }
136 
137  sn = mSlDevPid.begin()->first;
138  sn_str = std::to_string(sn);
139  }
140 
141  mDevSerial = sn;
142 
143  std::wstring wide_string = std::wstring(sn_str.begin(), sn_str.end());
144  const wchar_t* wsn = wide_string.c_str();
145 
146  uint16_t pid = mSlDevPid[sn];
147 
148  mDevHandle = hid_open(SL_USB_VENDOR, pid, wsn );
149 
150  if (!mDevHandle)
151  {
152  std::string msg = "Connection to device with sn ";
153  msg += sn_str;
154  msg += " failed";
155 
156  ERROR_OUT(mVerbose,msg);
157 
158  mDevFwVer = -1;
159  mDevSerial = -1;
160 
161  return false;
162  }
163 
164  if(mVerbose)
165  {
166  std::string msg = "Connected to device with sn ";
167  msg += sn_str;
168 
169  INFO_OUT(mVerbose,msg);
170  }
171 
172  mDevFwVer = mSlDevFwVer[sn];
173  mDevPid = pid;
174  mInitialized = startCapture();
175 
176  return true;

294 void SensorCapture::grabThreadFunc()
295 {
296  mStopCapture = false;
297  mGrabRunning = false;
298 
299  mNewIMUData=false;
300  mNewMagData=false;
301  mNewEnvData=false;
302  mNewCamTempData=false;
303 
304  // Read sensor data
305  unsigned char usbBuf[65];
306 
307  int ping_data_count = 0;
308 
309  mFirstImuData = true;
310 
311  uint64_t rel_mcu_ts = 0;
312 
313  mSysTsQueue.reserve(TS_SHIFT_VAL_COUNT);
314  mMcuTsQueue.reserve(TS_SHIFT_VAL_COUNT);
315 
316  while (!mStopCapture)
317  {
318  // ----> Keep data stream alive
319  // sending a ping aboutonce per second
320  // to keep the streaming alive
321  if(ping_data_count>=400) {
322  ping_data_count=0;
323  sendPing();
324  };
325  ping_data_count++;
326  // <---- Keep data stream alive
327 
328  mGrabRunning=true;
329 
330  // Sensor data request
331  usbBuf[1]=REP_ID_SENSOR_DATA;
332  int res = hid_read_timeout( mDevHandle, usbBuf, 64, 2000 );
333 
334  // ----> Data received?
335  if( res < static_cast<int>(sizeof(SensData)) ) {
336  hid_set_nonblocking( mDevHandle, 0 );
337  continue;
338  }
339  // <---- Data received?
340 
341  // ----> Received data are correct?
342  int target_struct_id = 0;
343  if (mDevPid==SL_USB_PROD_MCU_ZED2_REVA)
344  target_struct_id = REP_ID_SENSOR_DATA;
345 
346  if( usbBuf[0] != target_struct_id)
347  {
348  if(mVerbose)
349  {
350  WARNING_OUT(mVerbose,std::string("REP_ID_SENSOR_DATA - Sensor Data type mismatch") );
351  }
352 
353  hid_set_nonblocking( mDevHandle, 0 );
354  continue;
355  }
356  // <---- Received data are correct?
357 
358  // Data structure static conversion
359  SensData* data = (SensData*)usbBuf;
360 
361  // ----> Timestamp update
362  uint64_t mcu_ts_nsec = static_cast<uint64_t>(std::round(static_cast<float>(data->timestamp)*TS_SCALE));
363 
364  if(mFirstImuData && data->imu_not_valid!=1)
365  {
366  mStartSysTs = getWallTimestamp(); // Starting system timestamp
367  //std::cout << "SensorCapture: " << mStartSysTs << std::endl;
368 
369  mLastMcuTs = mcu_ts_nsec;
370  mFirstImuData = false;
371  continue;
372  }
373 
374  uint64_t delta_mcu_ts_raw = mcu_ts_nsec - mLastMcuTs;
375 
376  //std::cout << "Internal MCU freq: " << 1e9/delta_mcu_ts_raw << " Hz" << std::endl;
377 
378  mLastMcuTs = mcu_ts_nsec;
379  // <---- Timestamp update
380 
381  // Apply timestamp drift scaling factor
382  rel_mcu_ts += static_cast<uint64_t>(static_cast<double>(delta_mcu_ts_raw)*mNTPTsScaling);
383 
384  // mStartSysTs is synchronized to Video TS when sync is enabled using \ref VideoCapture::enableSensorSync
385  uint64_t current_data_ts = (mStartSysTs-mSyncOffset) + rel_mcu_ts;
386 
387  // ----> Camera/Sensors Synchronization
388  if( data->sync_capabilities != 0 ) // Synchronization active
389  {
390  if(mLastFrameSyncCount!=0 && (data->frame_sync!=0 || data->frame_sync_count>mLastFrameSyncCount))
391  {
392 #if 0 // Timestamp sync debug info
393  std::cout << "MCU sync information: " << std::endl;
394  std::cout << " * data->frame_sync: " << (int)data->frame_sync << std::endl;
395  std::cout << " * data->frame_sync_count: " << data->frame_sync_count << std::endl;
396  std::cout << " * mLastFrameSyncCount: " << mLastFrameSyncCount << std::endl;
397  std::cout << " * MCU timestamp scaling: " << mNTPTsScaling << std::endl;
398 #endif
399  mSysTsQueue.push_back( getSteadyTimestamp() ); // Steady host timestamp
400  mMcuTsQueue.push_back( current_data_ts ); // MCU timestamp
401 
402  // Once we have enough data, calculate the drift scaling factor
403  if (mSysTsQueue.size()==TS_SHIFT_VAL_COUNT && mMcuTsQueue.size() == TS_SHIFT_VAL_COUNT)
404  {
405  //First and last ts
406  int first_index = 5;
407  if (mNTPAdjustedCount <= NTP_ADJUST_CT) {
408  first_index = TS_SHIFT_VAL_COUNT/2;
409  }
410 
411  uint64_t first_ts_imu = mMcuTsQueue.at(first_index);
412  uint64_t last_ts_imu = mMcuTsQueue.at(mMcuTsQueue.size()-1);
413  uint64_t first_ts_cam = mSysTsQueue.at(first_index);
414  uint64_t last_ts_cam = mSysTsQueue.at(mSysTsQueue.size()-1);
415  double scale = double(last_ts_cam-first_ts_cam) / double(last_ts_imu-first_ts_imu);
416  //CLAMP
417  if (scale > 1.2) scale = 1.2;
418  if (scale < 0.8) scale = 0.8;
419 
420  //Adjust scaling continuoulsy. No jump so that ts(n) - ts(n-1) == 400Hz
421  mNTPTsScaling*=scale;
422 
423  //scale will be applied to the next values, so clear the vector and wait until we have enough data again
424  mMcuTsQueue.clear();
425  mSysTsQueue.clear();
426 
427  // Count the number of completed time shift factor estimations
428  mNTPAdjustedCount++;
429 
430 #ifdef VIDEO_MOD_AVAILABLE
431  // ----> Signal update offset to VideoCapture
432  if(mVideoPtr)
433  {
434  mSyncTs = current_data_ts;
435  mVideoPtr->setReadyToSync();
436  }
437  // <---- Update offset
438 #endif //VIDEO_MOD_AVAILABLE
439  }
440  }
441  }
442  mLastFrameSyncCount = data->frame_sync_count;
443  // <---- Camera/Sensors Synchronization
444 
445  // ----> IMU data
446  mIMUMutex.lock();
447  mLastIMUData.sync = data->frame_sync;
448  mLastIMUData.valid = data->imu_not_valid!=1;
449  mLastIMUData.timestamp = current_data_ts;
450  mLastIMUData.aX = data->aX*ACC_SCALE;
451  mLastIMUData.aY = data->aY*ACC_SCALE;
452  mLastIMUData.aZ = data->aZ*ACC_SCALE;
453  mLastIMUData.gX = data->gX*GYRO_SCALE;
454  mLastIMUData.gY = data->gY*GYRO_SCALE;
455  mLastIMUData.gZ = data->gZ*GYRO_SCALE;
456  mLastIMUData.temp = data->imu_temp*TEMP_SCALE;
457  mNewIMUData = true;
458  mIMUMutex.unlock();
459 
460  //std::string msg = std::to_string(mLastMAGData.timestamp);
461  //INFO_OUT(msg);
462  // <---- IMU data
463 
464  // ----> Magnetometer data
465  if(data->mag_valid == SensMagData::NEW_VAL)
466  {
467  mMagMutex.lock();
468  mLastMagData.valid = SensMagData::NEW_VAL;
469  mLastMagData.timestamp = current_data_ts;
470  mLastMagData.mY = data->mY*MAG_SCALE;
471  mLastMagData.mZ = data->mZ*MAG_SCALE;
472  mLastMagData.mX = data->mX*MAG_SCALE;
473  mNewMagData = true;
474  mMagMutex.unlock();
475 
476  //std::string msg = std::to_string(mLastMAGData.timestamp);
477  //INFO_OUT(msg);
478  }
479  else
480  {
481  mLastIMUData.valid = static_cast<SensMagData::MagStatus>(data->mag_valid);
482  }
483  // <---- Magnetometer data
484 
485  // ----> Environmental data
486  if(data->env_valid == SensEnvData::NEW_VAL)
487  {
488  mEnvMutex.lock();
489  mLastEnvData.valid = SensEnvData::NEW_VAL;
490  mLastEnvData.timestamp = current_data_ts;
491  mLastEnvData.temp = data->temp*TEMP_SCALE;
492  if( atLeast(mDevFwVer, ZED_2_FW::FW_3_9))
493  {
494  mLastEnvData.press = data->press*PRESS_SCALE_NEW;
495  mLastEnvData.humid = data->humid*HUMID_SCALE_NEW;
496  }
497  else
498  {
499  mLastEnvData.press = data->press*PRESS_SCALE_OLD;
500  mLastEnvData.humid = data->humid*HUMID_SCALE_OLD;
501  }
502  mNewEnvData = true;
503  mEnvMutex.unlock();
504 
505  //std::string msg = std::to_string(mLastENVData.timestamp);
506  //INFO_OUT(msg);
507  }
508  else
509  {
510  mLastIMUData.valid = static_cast<SensMagData::MagStatus>(data->mag_valid);
511  }
512  // <---- Environmental data
513 
514  // ----> Camera sensors temperature data
515  if(data->temp_cam_left != TEMP_NOT_VALID &&
516  data->temp_cam_left != TEMP_NOT_VALID &&
517  data->env_valid == SensEnvData::NEW_VAL ) // Sensor temperature is linked to Environmental data acquisition at FW level
518  {
519  mCamTempMutex.lock();
520  mLastCamTempData.valid = true;
521  mLastCamTempData.timestamp = current_data_ts;
522  mLastCamTempData.temp_left = data->temp_cam_left*TEMP_SCALE;
523  mLastCamTempData.temp_right = data->temp_cam_right*TEMP_SCALE;
524  mNewCamTempData=true;
525  mCamTempMutex.unlock();
526 
527  //std::string msg = std::to_string(mLastCamTempData.timestamp);
528  //INFO_OUT(msg);
529  }
530  else
531  {
532  mLastCamTempData.valid = false;
533  }
534  // <---- Camera sensors temperature data
535  }
536 
537  mGrabRunning = false;
538 }
